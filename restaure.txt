    <script>
        const socket = io();
        let tempChart, vibChart, currentChart;

        const SEUILS = {
            temperature: { normal: 70, alerte: 75, critique: 85 },
            vibration: { normal: 600, alerte: 700, critique: 800 },
            courant: { normal: 3.0, alerte: 3.5, critique: 4.0 }
        };

        function initCharts() {
            const commonOptions = {
                responsive: true,
                maintainAspectRatio: true,
                plugins: {
                    legend: { labels: { color: '#a0a3c4', font: { size: 11 } } },
                },
                scales: {
                    y: { grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#a0a3c4' } },
                    x: { ticks: { color: '#a0a3c4', maxRotation: 45, autoSkip: true } }
                }
            };

            tempChart = new Chart(document.getElementById('tempChart'), {
                type: 'line',
                data: { labels: [], datasets: [{ label: '🌡️ Température (°C)', data: [], borderColor: '#ff6b6b', backgroundColor: 'rgba(255,107,107,0.05)', borderWidth: 2, pointRadius: 3, tension: 0.3, fill: true }] },
                options: { ...commonOptions, plugins: { ...commonOptions.plugins, tooltip: { callbacks: { label: (ctx) => `${ctx.raw}°C` } } } }
            });

            vibChart = new Chart(document.getElementById('vibChart'), {
                type: 'line',
                data: { labels: [], datasets: [{ label: '📳 Vibration', data: [], borderColor: '#4ecdc4', backgroundColor: 'rgba(78,205,196,0.05)', borderWidth: 2, pointRadius: 3, tension: 0.3, fill: true }] },
                options: commonOptions
            });

            currentChart = new Chart(document.getElementById('currentChart'), {
                type: 'line',
                data: { labels: [], datasets: [{ label: '⚡ Courant (A)', data: [], borderColor: '#45b7d1', backgroundColor: 'rgba(69,183,209,0.05)', borderWidth: 2, pointRadius: 3, tension: 0.3, fill: true }] },
                options: commonOptions
            });
        }

        function updateStatusIndicator(value, elementId, seuils) {
            const el = document.getElementById(elementId);
            if (value >= seuils.critique) {
                el.innerHTML = '🚨 CRITIQUE';
                el.style.background = 'rgba(231,76,60,0.2)';
                el.style.color = '#e74c3c';
            } else if (value >= seuils.alerte) {
                el.innerHTML = '⚠️ ALERTE';
                el.style.background = 'rgba(243,156,18,0.2)';
                el.style.color = '#f39c12';
            } else if (value > seuils.normal) {
                el.innerHTML = '⚠️ ATTENTION';
                el.style.background = 'rgba(243,156,18,0.15)';
                el.style.color = '#f1c40f';
            } else {
                el.innerHTML = '✓ Normal';
                el.style.background = 'rgba(46,204,113,0.15)';
                el.style.color = '#2ecc71';
            }
        }

        function updateDisplay(current, alarmes, niveau) {
            console.log('Mise à jour affichage:', current);
            
            if (current.temperature !== undefined) {
                document.getElementById('temperature').innerHTML = current.temperature.toFixed(1);
                updateStatusIndicator(current.temperature, 'temp-status', SEUILS.temperature);
            }
            if (current.vibration !== undefined) {
                document.getElementById('vibration').innerHTML = current.vibration;
                updateStatusIndicator(current.vibration, 'vib-status', SEUILS.vibration);
            }
            if (current.courant !== undefined) {
                document.getElementById('courant').innerHTML = current.courant.toFixed(2);
                updateStatusIndicator(current.courant, 'current-status', SEUILS.courant);
            }
            if (current.timestamp) {
                document.getElementById('last-update').innerHTML = `Dernière mise à jour : ${current.timestamp}`;
            }

            const statusCard = document.getElementById('status-card');
            const statusMsg = document.getElementById('status-message');
            statusCard.classList.remove('status-normal', 'status-alerte', 'status-critique');
            
            if (niveau === 'CRITIQUE') {
                statusCard.classList.add('status-critique');
                statusMsg.innerHTML = '🔴 État CRITIQUE – Intervention immédiate requise !';
            } else if (niveau === 'ATTENTION' || (alarmes && alarmes.length > 0)) {
                statusCard.classList.add('status-alerte');
                statusMsg.innerHTML = '🟠 État ALERTE – Surveillance renforcée nécessaire';
            } else {
                statusCard.classList.add('status-normal');
                statusMsg.innerHTML = '🟢 État NORMAL – Fonctionnement correct';
            }

            const alertesList = document.getElementById('alertes-list');
            if (alarmes && alarmes.length > 0) {
                alertesList.innerHTML = alarmes.map(a => `<li style="border-left-color: ${a.includes('CRITIQUE') ? '#e74c3c' : '#f39c12'}">⚠️ ${a}</li>`).join('');
            } else {
                alertesList.innerHTML = '<li>✅ Aucune alerte – Machine en bon état</li>';
            }
        }

        function updateCharts(historique) {
            if (historique && historique.length > 0) {
                const labels = historique.map(h => h.timestamp);
                
                tempChart.data.labels = labels;
                tempChart.data.datasets[0].data = historique.map(h => h.temperature);
                tempChart.update();
                
                vibChart.data.labels = labels;
                vibChart.data.datasets[0].data = historique.map(h => h.vibration);
                vibChart.update();
                
                currentChart.data.labels = labels;
                currentChart.data.datasets[0].data = historique.map(h => h.courant);
                currentChart.update();
            }
        }

        // Socket.IO événements
        socket.on('connect', () => {
            console.log('✅ Connecté au serveur');
            document.getElementById('connectionBadge').innerHTML = '<i class="fas fa-sync-alt"></i> Temps réel | Connecté';
        });

        socket.on('initial_data', (data) => {
            console.log('📦 Données initiales reçues:', data);
            if (data.current && data.current.temperature !== undefined && data.current.temperature !== 0) {
                updateDisplay(data.current, data.alarmes || [], data.niveau || data.current.etat || 'NORMAL');
                if (data.historique) updateCharts(data.historique);
            }
        });

        socket.on('new_data', (data) => {
            console.log('📊 Nouvelles données reçues:', data);
            if (data.current) {
                updateDisplay(data.current, data.alarmes || [], data.niveau || data.current.etat || 'NORMAL');
                if (data.historique) updateCharts(data.historique);
            }
        });

        socket.on('disconnect', () => {
            console.log('❌ Déconnecté du serveur');
            document.getElementById('connectionBadge').innerHTML = '<i class="fas fa-sync-alt"></i> Déconnecté';
        });

        // Initialisation
        initCharts();
    </script>
